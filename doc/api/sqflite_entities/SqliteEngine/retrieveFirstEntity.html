Future<T?> retrieveFirstEntity<T>({
  Transaction? transaction,
}) async {
  final adapter = _adaptersMap[T]!;

  final maps =
      await (transaction ?? _database)!.query(adapter.tableName, limit: 1);

  if (maps.isEmpty) {
    return null;
  }

  return adapter.deserialize(maps[0]) as T;
}
